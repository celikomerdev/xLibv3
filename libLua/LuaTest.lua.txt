local UnityEngine = CS.UnityEngine
local xLib = CS.xLib
local Debug = UnityEngine.Debug
local xValueClass = xLib.xValueClass
local ValueFloat = xLib.xValueClass.ValueFloat







--return object with key
function Call()
	Debug.Log("Call")
	ValueFloat.GetGlobal("DefaultGlobal").Value = 8
	ValueFloatListener{item=ValueFloat.GetGlobal("DefaultGlobal"),register=true,onRegister=true,call=ListenerItem}
	ValueFloat.GetGlobal("DefaultGlobal").Value = 9
	ValueFloatListener{item=ValueFloat.GetGlobal("DefaultGlobal"),register=false,onRegister=true,call=ListenerItem}
	ValueFloat.GetGlobal("DefaultGlobal").Value = 10
end


function ValueFloatListener(table)
	setmetatable(table,{__index={viewId="",order=0,onRegister=false,worker=nil}})
	table.item:Listener(table.register,table.call,table.viewId,table.order,table.onRegister,table.worker)
end


function ListenerItem(valueCall)
	Debug.Log("Listener:"..valueCall)
end
